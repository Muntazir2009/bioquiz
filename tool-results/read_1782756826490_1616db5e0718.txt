  5296→function updateBadges(){
  5297→  // v29: derive unread truth from dmMeta (not the locally cached dmUnread
  5298→  // counter, which can drift). For each thread sum meta.unread[uid] but
  5299→  // discard stale counts where lastTs is older than our last read marker.
  5300→  let dmTotal=0;
  5301→  try{
  5302→    Object.keys(dmMeta||{}).forEach(did=>{
  5303→      const m=dmMeta[did]; if(!m||!m.unread) return;
  5304→      const n=Number(m.unread[uid]||0); if(!n||n<=0) return;
  5305→      // Skip the conversation we're actively reading.
  5306→      if(activeView==='dmconv' && did===activeDmId) return;
  5307→      const lastTs=Number(m.lastTs||0);
  5308→      const myRead=Number((dmReadCache && dmReadCache[did] && dmReadCache[did][uid])||0);
  5309→      if(lastTs && myRead && lastTs<=myRead){
  5310→        // Stale counter — heal it in the background.
  5311→        if(db) db.ref('bq_dms/'+did+'/meta/unread/'+uid).set(0).catch(()=>{});
  5312→        return;
  5313→      }
  5314→      dmTotal+=n;
  5315→      dmUnread[did]=n;
  5316→    });
  5317→  }catch(_){ dmTotal=Object.values(dmUnread).reduce((s,n)=>s+n,0); }
  5318→  const nb=document.getElementById('bqdmnb');
  5319→  if(nb){if(dmTotal>0){nb.textContent=dmTotal>9?'9+':dmTotal;nb.classList.add('show');}else nb.classList.remove('show');}
  5320→  const total=gUnread+dmTotal;
  5321→  const badge=document.getElementById('bqbadge');
  5322→  if(!badge)return;
  5323→  if(total>0){badge.textContent=total>9?'9+':total;badge.classList.add('show');}else badge.classList.remove('show');
  5324→}
